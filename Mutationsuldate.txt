local function writefile_safely(filename, content)
    if writefile then
        pcall(function() writefile(filename, content) end)
    end
end

local function readfile_safely(filename)
    if readfile then
        local success, result = pcall(function() return readfile(filename) end)
        if success then return result end
    end
    return nil
end

    --// TARGETING LOGIC
    local function get_best_target()
        local target, dist = nil, Settings.FOV
        local center = Vector2.new(camera.ViewportSize.X / 2, camera.ViewportSize.Y / 2)
        
        local chance = math.random(1, 100)
        local currentPart = (chance <= Settings.HeadshotRate) and "Head" or "UpperTorso"
        
        for _, v in pairs(players:GetPlayers()) do
            if v ~= lp and v.Character and v.Character:FindFirstChild(currentPart) then
                local hitPart = v.Character[currentPart]
                local pos, onScreen = camera:WorldToViewportPoint(hitPart.Position)
                if onScreen then
                    local mag = (Vector2.new(pos.X, pos.Y) - center).Magnitude
                    if mag < dist then
                        local isVisible = true
                        if Settings.WallCheck then
                            local origin = camera.CFrame.Position
                            local direction = (hitPart.Position - origin).Unit * (hitPart.Position - origin).Magnitude
                            local ray = Ray.new(origin, direction)
                            local ignoreList = {lp.Character, v.Character, camera}
                            local hit = workspace:FindPartOnRayWithIgnoreList(ray, ignoreList)
                            if hit then
                                isVisible = false
                            end
                        end
                        
                        if isVisible then
                            target = hitPart
                            dist = mag
                        end
                    end
                end
            end
        end
        return target
    end

        valLabel.Size = UDim2.new(0.3, 0, 0, 25)
        valLabel.TextColor3 = Color3.fromRGB(150, 120, 240)
        valLabel.Font = Enum.Font.GothamBold
        valLabel.TextSize = 12
        valLabel.TextXAlignment = Enum.TextXAlignment.Right
        valLabel.BackgroundTransparency = 1

        local bar = Instance.new("TextButton", sFrame)
        bar.Size = UDim2.new(1, -24, 0, 6)
        bar.Position = UDim2.new(0, 12, 0, 40)
        bar.BackgroundColor3 = Color3.fromRGB(45, 40, 55)
        bar.Text = ""
        Instance.new("UICorner", bar).CornerRadius = UDim.new(1, 0)
        
        local fill = Instance.new("Frame", bar)
        fill.Size = UDim2.new((default - min) / (max - min), 0, 1, 0)
        fill.BackgroundColor3 = Color3.fromRGB(140, 0, 255)
        Instance.new("UICorner", fill).CornerRadius = UDim.new(1, 0)

        local dot = Instance.new("Frame", bar)
        dot.Position = UDim2.new((default - min) / (max - min), -6, 0.5, -6)
        dot.Size = UDim2.new(0, 12, 0, 12)
        dot.BackgroundColor3 = Color3.fromRGB(255, 255, 255)
        Instance.new("UICorner", dot).CornerRadius = UDim.new(1, 0)
        
        local dotStroke = Instance.new("UIStroke", dot)
        dotStroke.Color = Color3.fromRGB(140, 0, 255)
        dotStroke.Thickness = 2

        local dragging = false
        local lastVal = default

        local function update_slider(input)
            local inputPos = input.Position.X
            local barAbsPos = bar.AbsolutePosition.X
            local barAbsSize = bar.AbsoluteSize.X
            local percentage = math.clamp((inputPos - barAbsPos) / barAbsSize, 0, 1)
            
            local value = math.floor(min + (max - min) * percentage)
            valLabel.Text = tostring(value) .. "/" .. tostring(max)
            
            fill.Size = UDim2.new(percentage, 0, 1, 0)
            dot.Position = UDim2.new(percentage, -6, 0.5, -6)
            
            if value ~= lastVal then
                lastVal = value
                play_tick()
            end
            
            callback(value)
        end

        bar.InputBegan:Connect(function(input)
            if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
                dragging = true
                update_slider(input)
                ts:Create(sStroke, tweenInfo, {Color = Color3.fromRGB(140, 0, 255)}):Play()
            end
        end)

        uis.InputChanged:Connect(function(input)
            if dragging and (input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch) then
                update_slider(input)
            end
        end)

        uis.InputEnded:Connect(function(input)
            if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
                if dragging then
                    dragging = false
                    ts:Create(sStroke, tweenInfo, {Color = Color3.fromRGB(35, 30, 45)}):Play()
                end
            end
        end)
    end
